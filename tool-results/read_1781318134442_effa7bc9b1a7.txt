    26‚Üí/**
    27‚Üí * Map a string value to the SearchIntent enum.
    28‚Üí * Falls back to UNKNOWN if the value is not a valid enum member.
    29‚Üí */
    30‚Üífunction toSearchIntent(value?: string): string {
    31‚Üí  const valid = [
    32‚Üí    'INFORMATIONAL',
    33‚Üí    'NAVIGATIONAL',
    34‚Üí    'TRANSACTIONAL',
    35‚Üí    'COMMERCIAL_INVESTIGATION',
    36‚Üí    'LOCAL',
    37‚Üí    'BRANDED',
    38‚Üí    'UNKNOWN',
    39‚Üí  ];
    40‚Üí  if (!value) return 'UNKNOWN';
    41‚Üí  const upper = value.toUpperCase().replace(/\s+/g, '_');
    42‚Üí  return valid.includes(upper) ? upper : 'UNKNOWN';
    43‚Üí}
    44‚Üí
    45‚Üí/**
    46‚Üí * Map a string value to the FunnelStage enum.
    47‚Üí * Falls back to UNKNOWN if the value is not a valid enum member.
    48‚Üí */
    49‚Üífunction toFunnelStage(value?: string): string {
    50‚Üí  const valid = ['AWARENESS', 'CONSIDERATION', 'DECISION', 'RETENTION', 'UNKNOWN'];
    51‚Üí  if (!value) return 'UNKNOWN';
    52‚Üí  const upper = value.toUpperCase().replace(/\s+/g, '_');
    53‚Üí  return valid.includes(upper) ? upper : 'UNKNOWN';
    54‚Üí}
    55‚Üí
    56‚Üí/**
    57‚Üí * Map a string value to the ContentApprovalStatus enum.
    58‚Üí * Falls back to DRAFT if the value is not a valid enum member.
    59‚Üí */
    60‚Üífunction toApprovalStatus(value?: string): string {
    61‚Üí  const valid = ['DRAFT', 'IN_REVIEW', 'APPROVED', 'PUBLISHED', 'ARCHIVED'];
    62‚Üí  if (!value) return 'DRAFT';
    63‚Üí  const upper = value.toUpperCase();
    64‚Üí  return valid.includes(upper) ? upper : 'DRAFT';
    65‚Üí}

   286‚Üí *

   289‚Üí * @returns Paginated list of brief summaries
   290‚Üí */
   291‚Üíexport async function listBriefs(
   292‚Üí  projectId: string,
   293‚Üí  filters?: BriefFilters
   294‚Üí): Promise<PaginatedBriefs> {
   295‚Üí  const page = Math.max(1, filters?.page ?? 1);
   296‚Üí  const pageSize = Math.min(100, Math.max(1, filters?.pageSize ?? 20));
   297‚Üí  const sortBy = filters?.sortBy ?? 'createdAt';
   298‚Üí  const sortDirection = filters?.sortDirection ?? 'desc';
   299‚Üí
   300‚Üí  // Build where clause
   301‚Üí  const where: Record<string, unknown> = { projectId };
   302‚Üí
   303‚Üí  if (filters?.approvalStatus) {
   304‚Üí    where.approvalStatus = toApprovalStatus(filters.approvalStatus);
   305‚Üí  }
   306‚Üí
   307‚Üí  if (filters?.searchIntent) {
   308‚Üí    where.searchIntent = toSearchIntent(filters.searchIntent);
   309‚Üí  }
   310‚Üí
   311‚Üí  if (filters?.funnelStage) {
   312‚Üí    where.funnelStage = toFunnelStage(filters.funnelStage);
   313‚Üí  }
   314‚Üí
   315‚Üí  if (filters?.search) {

   317‚Üí      { title: { contains: filters.search } },
   318‚Üí      { targetKeyword: { contains: filters.search } },
   319‚Üí    ];
   320‚Üí  }
   321‚Üí
   322‚Üí  // Get total count
   323‚Üí  const total = await db.contentBrief.count({ where });
   324‚Üí
   325‚Üí  // Get paginated results
   326‚Üí  const briefs = await db.contentBrief.findMany({
   327‚Üí    where,
   328‚Üí    select: {
   329‚Üí      id: true,
   330‚Üí      title: true,
   331‚Üí      targetKeyword: true,
   332‚Üí      searchIntent: true,
   333‚Üí      funnelStage: true,
   334‚Üí      approvalStatus: true,
   335‚Üí      targetWordCount: true,
   336‚Üí      createdAt: true,
   337‚Üí      updatedAt: true,
   338‚Üí      versions: { select: { id: true } },
   339‚Üí    },
   340‚Üí    orderBy: { [sortBy]: sortDirection },
   341‚Üí    skip: (page - 1) * pageSize,
   342‚Üí    take: pageSize,
   343‚Üí  });
   344‚Üí
   345‚Üí  const items: BriefSummary[] = briefs.map((b) => ({
   346‚Üí    id: b.id,

   349‚Üí    searchIntent: b.searchIntent,
   350‚Üí    funnelStage: b.funnelStage,
   351‚Üí    approvalStatus: b.approvalStatus,
   352‚Üí    targetWordCount: b.targetWordCount ?? undefined,
   353‚Üí    versionCount: b.versions.length,
   354‚Üí    createdAt: b.createdAt,
   355‚Üí    updatedAt: b.updatedAt,
   356‚Üí  }));
   357‚Üí
   358‚Üí  return {
   359‚Üí    items,
   360‚Üí    total,
   361‚Üí    page,
   362‚Üí    pageSize,
   363‚Üí    totalPages: Math.ceil(total / pageSize),

   365‚Üí}
