    47‚Üí/**
    48‚Üí * Calculate a simple n-gram similarity score between two strings.
    49‚Üí * Uses word-level trigrams for comparison. Returns a value between 0 and 1.
    50‚Üí *
    51‚Üí * @param textA - First text
    52‚Üí * @param textB - Second text
    53‚Üí * @returns Similarity score (0 = completely different, 1 = identical)
    54‚Üí */
    55‚Üífunction calculateSimilarity(textA: string, textB: string): number {
    56‚Üí  const normalize = (text: string) =>
    57‚Üí    text
    58‚Üí      .toLowerCase()
    59‚Üí      .replace(/[^\w\s]/g, '')
    60‚Üí      .split(/\s+/)
    61‚Üí      .filter((w) => w.length > 2);
    62‚Üí
    63‚Üí  const wordsA = normalize(textA);
    64‚Üí  const wordsB = normalize(textB);
    65‚Üí
    66‚Üí  if (wordsA.length === 0 || wordsB.length === 0) return 0;
    67‚Üí  if (wordsA.join(' ') === wordsB.join(' ')) return 1;
    68‚Üí
    69‚Üí  // Generate trigrams
    70‚Üí  const getTrigrams = (words: string[]): Set<string> => {
    71‚Üí    const trigrams = new Set<string>();
    72‚Üí    for (let i = 0; i < words.length - 2; i++) {
    73‚Üí      trigrams.add(`${words[i]} ${words[i + 1]} ${words[i + 2]}`);
    74‚Üí    }
    75‚Üí    return trigrams;
    76‚Üí  };
    77‚Üí
    78‚Üí  const trigramsA = getTrigrams(wordsA);
    79‚Üí  const trigramsB = getTrigrams(wordsB);
    80‚Üí
    81‚Üí  if (trigramsA.size === 0 || trigramsB.size === 0) {
    82‚Üí    // Fall back to unigram Jaccard similarity
    83‚Üí    const setA = new Set(wordsA);
    84‚Üí    const setB = new Set(wordsB);
    85‚Üí    const intersection = new Set([...setA].filter((w) => setB.has(w)));
    86‚Üí    const union = new Set([...setA, ...setB]);
    87‚Üí    return union.size > 0 ? intersection.size / union.size : 0;
    88‚Üí  }
    89‚Üí
    90‚Üí  const intersection = new Set([...trigramsA].filter((t) => trigramsB.has(t)));
    91‚Üí  const union = new Set([...trigramsA, ...trigramsB]);
    92‚Üí  return union.size > 0 ? intersection.size / union.size : 0;
    93‚Üí}

   100‚Üífunction countWords(text: string): number {
   101‚Üí  return text
   102‚Üí    .replace(/[#*_\[\]()]/g, ' ')
   103‚Üí    .split(/\s+/)
   104‚Üí    .filter((w) => w.length > 0).length;
   105‚Üí}

   111‚Üí/**
   112‚Üí * Gate 1: Unique Data Requirement
   113‚Üí *
   114‚Üí * Each generated page must use distinct data values. Checks that key
   115‚Üí * variables differ between pages by comparing the current row data
   116‚Üí * against previously generated pages for this template.
   117‚Üí */
   118‚Üíasync function checkUniqueData(
   119‚Üí  templateId: string,
   120‚Üí  rowData: Record<string, unknown>,
   121‚Üí  keyVariables: string[]
   122‚Üí): Promise<QualityGateResult> {
   123‚Üí  // Get existing pages for this template to compare data
   124‚Üí  const existingPages = await db.programmaticPage.findMany({
   125‚Üí    where: {
   126‚Üí      templateId,
   127‚Üí      deletedAt: null,
   128‚Üí      status: { in: ['APPROVED', 'PENDING', 'QUALITY_CHECK'] },
   129‚Üí    },
   130‚Üí    select: { rowData: true },
   131‚Üí  });
   132‚Üí
   133‚Üí  if (existingPages.length === 0) {
   134‚Üí    return {
   135‚Üí      gateName: QUALITY_GATES.UNIQUE_DATA.name,
   136‚Üí      passed: true,
   137‚Üí      score: 100,
   138‚Üí      message: 'Eerste pagina voor dit sjabloon ‚Äî geen vergelijking nodig.',
   139‚Üí    };
   140‚Üí  }
   141‚Üí
   142‚Üí  const rowValues = keyVariables.map((v) => String(rowData[v] ?? '')).filter((v) => v.length > 0);
   143‚Üí
   144‚Üí  if (rowValues.length === 0) {
   145‚Üí    return {
   146‚Üí      gateName: QUALITY_GATES.UNIQUE_DATA.name,
   147‚Üí      passed: false,
   148‚Üí      score: 0,
   149‚Üí      message: 'Geen sleutelwaarden gevonden in de gegevens om uniciteit te controleren.',
   150‚Üí    };
   151‚Üí  }
   152‚Üí
   153‚Üí  let duplicateCount = 0;
   154‚Üí  for (const page of existingPages) {
   155‚Üí    try {
   156‚Üí      const existingData: Record<string, unknown> = JSON.parse(page.rowData);
   157‚Üí      const existingValues = keyVariables.map((v) => String(existingData[v] ?? '')).filter((v) => v.length > 0);
   158‚Üí      // Check if all key variable values are the same
   159‚Üí      if (rowValues.length === existingValues.length && rowValues.every((v, i) => v === existingValues[i])) {
   160‚Üí        duplicateCount++;
   161‚Üí      }
   162‚Üí    } catch {
   163‚Üí      // Skip pages with invalid JSON data
   164‚Üí    }
   165‚Üí  }
   166‚Üí
   167‚Üí  if (duplicateCount > 0) {
   168‚Üí    return {
   169‚Üí      gateName: QUALITY_GATES.UNIQUE_DATA.name,
   170‚Üí      passed: false,
   171‚Üí      score: 0,
   172‚Üí      message: `De gegevens van deze pagina komen exact overeen met ${duplicateCount} bestaande pagina('s). Elke pagina moet unieke gegevens bevatten.`,
   173‚Üí      details: { duplicateCount, keyVariables },
   174‚Üí    };
   175‚Üí  }
   176‚Üí
   177‚Üí  // Partial uniqueness score ‚Äî check how many key variables have at least some variation
   178‚Üí  let variationScore = 100;
   179‚Üí  for (const page of existingPages) {
   180‚Üí    try {
   181‚Üí      const existingData: Record<string, unknown> = JSON.parse(page.rowData);
   182‚Üí      const similarity = keyVariables.filter(
   183‚Üí        (v) => String(rowData[v] ?? '') === String(existingData[v] ?? '')
   184‚Üí      ).length / keyVariables.length;
   185‚Üí      // Reduce score for partial matches
   186‚Üí      variationScore = Math.min(variationScore, Math.round((1 - similarity) * 100));
   187‚Üí    } catch {
   188‚Üí      // Skip invalid data
   189‚Üí    }
   190‚Üí  }
   191‚Üí
   192‚Üí  return {
   193‚Üí    gateName: QUALITY_GATES.UNIQUE_DATA.name,
   194‚Üí    passed: variationScore >= 50,
   195‚Üí    score: variationScore,
   196‚Üí    message: variationScore >= 50
   197‚Üí      ? 'Gegevens zijn voldoende uniek ten opzichte van bestaande pagina\'s.'
   198‚Üí      : 'Waarschuwing: gegevens lijken sterk op bestaande pagina\'s. Overweeg meer variatie toe te voegen.',
   199‚Üí    details: { keyVariables, variationScore },
   200‚Üí  };
   201‚Üí}

   203‚Üí/**
   204‚Üí * Gate 2: Minimum Value Threshold
   205‚Üí *
   206‚Üí * Page must have at least 300 words, a title, and at least 3 unique data points.
   207‚Üí */
   208‚Üífunction checkMinValueThreshold(
   209‚Üí  generatedContent: string,
   210‚Üí  rowData: Record<string, unknown>,
   211‚Üí  config: QualityGatesConfig
   212‚Üí): QualityGateResult {
   213‚Üí  const minWordCount = config.minWordCount ?? 300;
   214‚Üí  const minUniquePoints = config.minUniqueDataPoints ?? 3;
   215‚Üí
   216‚Üí  const wordCount = countWords(generatedContent);
   217‚Üí  const hasTitle = /^#\s+.+/m.test(generatedContent);
   218‚Üí  const uniqueDataPoints = Object.values(rowData).filter(
   219‚Üí    (v) => v !== undefined && v !== null && String(v).trim().length > 0
   220‚Üí  ).length;
   221‚Üí
   222‚Üí  const wordScore = Math.min(100, Math.round((wordCount / minWordCount) * 100));
   223‚Üí  const titleScore = hasTitle ? 100 : 0;
   224‚Üí  const dataPointsScore = Math.min(100, Math.round((uniqueDataPoints / minUniquePoints) * 100));
   225‚Üí
   226‚Üí  const overallScore = Math.round((wordScore + titleScore + dataPointsScore) / 3);
   227‚Üí  const passed = wordCount >= minWordCount && hasTitle && uniqueDataPoints >= minUniquePoints;
   228‚Üí
   229‚Üí  const issues: string[] = [];
   230‚Üí  if (wordCount < minWordCount) {
   231‚Üí    issues.push(`Woordenaantal (${wordCount}) is onder het minimum van ${minWordCount}.`);
   232‚Üí  }
   233‚Üí  if (!hasTitle) {
   234‚Üí    issues.push('De pagina heeft geen titel (H1-kop).');
   235‚Üí  }
   236‚Üí  if (uniqueDataPoints < minUniquePoints) {
   237‚Üí    issues.push(`Aantal unieke gegevenspunten (${uniqueDataPoints}) is onder het minimum van ${minUniquePoints}.`);
   238‚Üí  }
   239‚Üí
   240‚Üí  return {
   241‚Üí    gateName: QUALITY_GATES.MIN_VALUE.name,
   242‚Üí    passed,
   243‚Üí    score: overallScore,
   244‚Üí    message: passed
   245‚Üí      ? `Pagina voldoet aan de minimale waardedrempel: ${wordCount} woorden, ${uniqueDataPoints} gegevenspunten.`
   246‚Üí      : `Pagina voldoet niet aan de minimale waardedrempel: ${issues.join(' ')}`,
   247‚Üí    details: { wordCount, minWordCount, hasTitle, uniqueDataPoints, minUniquePoints, wordScore, titleScore, dataPointsScore },
   248‚Üí  };
   249‚Üí}

   251‚Üí/**
   252‚Üí * Gate 3: Duplicate Check (BLOCKING)
   253‚Üí *
   254‚Üí * Compare generated content against existing pages in the project.
   255‚Üí * Reject if >80% similar.
   256‚Üí */
   257‚Üíasync function checkDuplicate(
   258‚Üí  projectId: string,
   259‚Üí  generatedContent: string,
   260‚Üí  config: QualityGatesConfig
   261‚Üí): Promise<QualityGateResult> {
   262‚Üí  const maxSimilarity = config.maxSimilarityThreshold ?? 0.8;
   263‚Üí
   264‚Üí  // Get existing crawled pages from the project
   265‚Üí  const existingPages = await db.page.findMany({
   266‚Üí    where: {
   267‚Üí      projectId,

   279‚Üí
   280‚Üí  if (existingPages.length === 0) {

   300‚Üí
   301‚Üí  // Also check against other programmatic pages in the project

   322‚Üí    }
   323‚Üí  }
   324‚Üí
   325‚Üí  const score = Math.round((1 - maxFoundSimilarity) * 100);
   326‚Üí  const passed = maxFoundSimilarity < maxSimilarity;
   327‚Üí
   328‚Üí  return {
   329‚Üí    gateName: QUALITY_GATES.DUPLICATE.name,
   330‚Üí    passed,
   331‚Üí    score,
   332‚Üí    message: passed
   333‚Üí      ? `Inhoud is voldoende uniek (max gelijkenis: ${Math.round(maxFoundSimilarity * 100)}%).`
   334‚Üí      : `Inhoud is te vergelijkbaar met een bestaande pagina (${Math.round(maxFoundSimilarity * 100)}% gelijkenis). Drempel: ${Math.round(maxSimilarity * 100)}%.`,

   339‚Üí/**

   345‚Üíasync function checkCannibalisation(
   346‚Üí  projectId: string,
   347‚Üí  targetKeyword: string

   380‚Üí

   389‚Üí
   390‚Üí  // Partial overlap scores lower
   391‚Üí  const exactMatches = overlappingKeywords.filter(
   392‚Üí    (k) => k.keyword.toLowerCase().trim() === normalizedKeyword
   393‚Üí  );
   394‚Üí
   395‚Üí  const score = exactMatches.length > 0 ? 20 : Math.max(50, 100 - overlappingKeywords.length * 15);
   396‚Üí  const passed = exactMatches.length === 0;
   397‚Üí
   398‚Üí  return {
   399‚Üí    gateName: QUALITY_GATES.CANNIBALISATION.name,
   400‚Üí    passed,
   401‚Üí    score,
   402‚Üí    message: passed
   403‚Üí      ? `Waarschuwing: ${overlappingKeywords.length} overlappende trefwoorden gevonden, maar geen exacte match voor "${targetKeyword}".`
   404‚Üí      : `Kannibalisatierisico: exacte trefwoordmatch "${targetKeyword}" komt al voor op ${overlappingKeywords.length} pagina('s).`,
   405‚Üí    details: {
   406‚Üí      targetKeyword,
   407‚Üí      overlappingKeywords: overlappingKeywords.map((k) => k.keyword),
   408‚Üí      exactMatches: exactMatches.length,
   409‚Üí      competingUrls: overlappingKeywords.map((k) => k.currentUrl).filter(Boolean),
   410‚Üí    },
   411‚Üí  };
   412‚Üí}

   414‚Üí/**
   415‚Üí * Gate 5: Template Completeness (BLOCKING)
   416‚Üí *
   417‚Üí * All {{variable}} placeholders must be filled. No empty sections.
   418‚Üí */
   419‚Üífunction checkTemplateCompleteness(
   420‚Üí  generatedContent: string,
   421‚Üí  rowData: Record<string, unknown>
   422‚Üí): QualityGateResult {
   423‚Üí  // Check for unfilled placeholders
   424‚Üí  const placeholderRegex = /\{\{(\w+)\}\}/g;
   425‚Üí  const unfilledPlaceholders: string[] = [];
   426‚Üí  let match: RegExpExecArray | null;
   427‚Üí
   428‚Üí  while ((match = placeholderRegex.exec(generatedContent)) !== null) {
   429‚Üí    unfilledPlaceholders.push(match[1]);
   430‚Üí  }
   431‚Üí
   432‚Üí  // Check for empty sections (## heading followed by no content or only whitespace)
   433‚Üí  const emptySectionRegex = /^##\s+.+\n\s*(?=\n#|$)/gm;
   434‚Üí  const emptySections: string[] = [];
   435‚Üí  const sectionMatches = generatedContent.matchAll(/^##\s+(.+)$/gm);
   436‚Üí  for (const sectionMatch of sectionMatches) {
   437‚Üí    const sectionName = sectionMatch[1];
   438‚Üí    const startPos = sectionMatch.index! + sectionMatch[0].length;
   439‚Üí    const nextHeading = generatedContent.indexOf('\n#', startPos);
   440‚Üí    const sectionContent = generatedContent.slice(
   441‚Üí      startPos,
   442‚Üí      nextHeading === -1 ? undefined : nextHeading
   443‚Üí    ).trim();
   444‚Üí
   445‚Üí    if (sectionContent.length === 0) {
   446‚Üí      emptySections.push(sectionName);
   447‚Üí    }
   448‚Üí  }
   449‚Üí
   450‚Üí  // Check for empty data values
   451‚Üí  const emptyDataKeys = Object.entries(rowData)
   452‚Üí    .filter(([, v]) => v === undefined || v === null || String(v).trim().length === 0)
   453‚Üí    .map(([k]) => k);
   454‚Üí
   455‚Üí  const totalIssues = unfilledPlaceholders.length + emptySections.length + emptyDataKeys.length;
   456‚Üí  const passed = unfilledPlaceholders.length === 0 && emptySections.length === 0;
   457‚Üí
   458‚Üí  let score = 100;
   459‚Üí  if (unfilledPlaceholders.length > 0) score -= unfilledPlaceholders.length * 25;
   460‚Üí  if (emptySections.length > 0) score -= emptySections.length * 15;
   461‚Üí  if (emptyDataKeys.length > 0) score -= emptyDataKeys.length * 5;
   462‚Üí  score = Math.max(0, score);
   463‚Üí
   464‚Üí  const issues: string[] = [];
   465‚Üí  if (unfilledPlaceholders.length > 0) {
   466‚Üí    issues.push(`Niet-ingevulde plaatsaanduidingen: ${unfilledPlaceholders.join(', ')}.`);
   467‚Üí  }
   468‚Üí  if (emptySections.length > 0) {
   469‚Üí    issues.push(`Lege secties: ${emptySections.join(', ')}.`);
   470‚Üí  }
   471‚Üí  if (emptyDataKeys.length > 0) {
   472‚Üí    issues.push(`Lege gegevenswaarden: ${emptyDataKeys.join(', ')}.`);
   473‚Üí  }
   474‚Üí
   475‚Üí  return {
   476‚Üí    gateName: QUALITY_GATES.TEMPLATE_COMPLETENESS.name,
   477‚Üí    passed,
   478‚Üí    score,
   479‚Üí    message: passed
   480‚Üí      ? 'Sjabloon is volledig ingevuld ‚Äî geen lege secties of ontbrekende waarden.'
   481‚Üí      : `Sjabloon is niet volledig: ${issues.join(' ')}`,
   482‚Üí    details: { unfilledPlaceholders, emptySections, emptyDataKeys },
   483‚Üí  };
   484‚Üí}

   488‚Üí *
   489‚Üí * Content must not violate brand profile prohibited terms/claims.

   491‚Üíasync function checkBrandCompliance(
   492‚Üí  projectId: string,
   493‚Üí  generatedContent: string
   494‚Üí): Promise<QualityGateResult> {
   495‚Üí  const brandProfile = await db.brandProfile.findUnique({
   496‚Üí    where: { projectId },
   497‚Üí  });
   498‚Üí
   499‚Üí  if (!brandProfile) {
   500‚Üí    return {
   501‚Üí      gateName: QUALITY_GATES.BRAND_CHECK.name,
   502‚Üí      passed: true,
   503‚Üí      score: 100,

   512‚Üí    ? JSON.parse(brandProfile.prohibitedClaims)

   514‚Üí

   520‚Üí      foundProhibitedTerms.push(term);

   522‚Üí  }
   523‚Üí
   524‚Üí  const foundProhibitedClaims: string[] = [];

   527‚Üí      foundProhibitedClaims.push(claim);

   530‚Üí
   531‚Üí  const totalViolations = foundProhibitedTerms.length + foundProhibitedClaims.length;
   532‚Üí  const passed = totalViolations === 0;

   535‚Üí  if (foundProhibitedTerms.length > 0) score -= foundProhibitedTerms.length * 20;

   537‚Üí  score = Math.max(0, score);

   539‚Üí  const issues: string[] = [];
   540‚Üí  if (foundProhibitedTerms.length > 0) {
   541‚Üí    issues.push(`Verboden terminologie gevonden: ${foundProhibitedTerms.join(', ')}.`);
   542‚Üí  }

   544‚Üí    issues.push(`Verboden claims gevonden: ${foundProhibitedClaims.join(', ')}.`);

   546‚Üí
   547‚Üí  return {
   548‚Üí    gateName: QUALITY_GATES.BRAND_CHECK.name,
   549‚Üí    passed,
   550‚Üí    score,
   551‚Üí    message: passed
   552‚Üí      ? 'Inhoud voldoet aan de merkrichtlijnen.'
   553‚Üí      : `Inhoud schendt merkrichtlijnen: ${issues.join(' ')}`,

   558‚Üí/**
   559‚Üí * Gate 7: Claim Check
   560‚Üí *
   561‚Üí * Mark unsupported claims using claim markers from Phase 3.
   562‚Üí * Checks for superlative claims, unsubstantiated statistics, and
   563‚Üí * absolute statements that lack supporting evidence.
   564‚Üí */
   565‚Üífunction checkClaims(generatedContent: string): QualityGateResult {
   566‚Üí  // Patterns that often indicate unsubstantiated claims
   567‚Üí  const claimPatterns = [
   568‚Üí    { pattern: /de beste/i, label: 'Superlatief: "de beste"' },
   569‚Üí    { pattern: /nummer\s*1/i, label: 'Absolute bewering: "nummer 1"' },
   570‚Üí    { pattern: /100%/g, label: 'Absoluut percentage: "100%"' },
   571‚Üí    { pattern: /altijd/gi, label: 'Absolute bewering: "altijd"' },
   572‚Üí    { pattern: /nooit/gi, label: 'Absolute bewering: "nooit"' },
   573‚Üí    { pattern: /iedereen/gi, label: 'Algemene bewering: "iedereen"' },
   574‚Üí    { pattern: / gegarandeerd/gi, label: 'Garantieclaim: "gegarandeerd"' },
   575‚Üí    { pattern: /bewezen/gi, label: 'Bewijsclaim: "bewezen" zonder bron' },
   576‚Üí    { pattern: /wetenschappelijk\s+bewezen/gi, label: 'Wetenschappelijke claim zonder bron' },
   577‚Üí    { pattern: /enige\s+(oplossing|optie|keuze)/gi, label: 'Exclusiviteitsclaim' },
   578‚Üí    { pattern: /onovertroffen/gi, label: 'Superlatief: "onovertroffen"' },
   579‚Üí    { pattern: /uniek\s+in\s+de\s+wereld/gi, label: 'Exclusiviteitsclaim: "uniek in de wereld"' },
   580‚Üí  ];
   581‚Üí
   582‚Üí  const foundClaims: Array<{ label: string; count: number }> = [];
   583‚Üí
   584‚Üí  for (const { pattern, label } of claimPatterns) {
   585‚Üí    const matches = generatedContent.match(pattern);
   586‚Üí    if (matches && matches.length > 0) {
   587‚Üí      foundClaims.push({ label, count: matches.length });
   588‚Üí    }
   589‚Üí  }
   590‚Üí
   591‚Üí  const totalClaims = foundClaims.reduce((sum, c) => sum + c.count, 0);
   592‚Üí  const passed = totalClaims === 0;
   593‚Üí
   594‚Üí  // Score based on number of claims found
   595‚Üí  let score = 100;
   596‚Üí  if (totalClaims > 0) score = Math.max(0, 100 - totalClaims * 15);
   597‚Üí
   598‚Üí  return {
   599‚Üí    gateName: QUALITY_GATES.CLAIM_CHECK.name,
   600‚Üí    passed,
   601‚Üí    score,
   602‚Üí    message: passed
   603‚Üí      ? 'Geen onderbouwde claims gedetecteerd in de inhoud.'
   604‚Üí      : `${totalClaims} potenti√´le ononderbouwde claim(s) gevonden: ${foundClaims.map((c) => c.label).join(', ')}.`,
   605‚Üí    details: { foundClaims, totalClaims },
   606‚Üí  };
   607‚Üí}

   609‚Üí/**
   610‚Üí * Gate 8: Internal Link Check
   611‚Üí *
   612‚Üí * Content should include at least one internal link suggestion.
   613‚Üí */
   614‚Üíasync function checkInternalLinks(
   615‚Üí  projectId: string,
   616‚Üí  generatedContent: string
   617‚Üí): Promise<QualityGateResult> {
   618‚Üí  // Check for Markdown links [text](/path) or HTML links <a href="/...">
   619‚Üí  const markdownLinks = generatedContent.match(/\[([^\]]+)\]\((\/[^)]+)\)/g) || [];
   620‚Üí  const htmlLinks = generatedContent.match(/<a\s+href="\/[^"]*"/gi) || [];
   621‚Üí  const totalInternalLinks = markdownLinks.length + htmlLinks.length;

   627‚Üí      deletedAt: null,

   631‚Üí    take: 20,
   632‚Üí  });
   633‚Üí

   638‚Üí  if (hasInternalLinks) {
   639‚Üí    score = Math.min(100, 50 + totalInternalLinks * 25);
   640‚Üí  } else if (linkSuggestionAvailable) {
   641‚Üí    score = 30;
   642‚Üí  } else {
   643‚Üí    score = 80; // No pages to link to, so not penalized heavily
   644‚Üí  }
   645‚Üí
   646‚Üí  const passed = hasInternalLinks || !linkSuggestionAvailable;
   647‚Üí
   648‚Üí  // Build suggestion details
   649‚Üí  const suggestions = existingPages.slice(0, 5).map((p) => ({
   650‚Üí    url: p.url,
   651‚Üí    title: p.title ?? 'Zonder titel',
   652‚Üí  }));
   653‚Üí
   654‚Üí  return {
   655‚Üí    gateName: QUALITY_GATES.INTERNAL_LINK.name,
   656‚Üí    passed,
   657‚Üí    score,
   658‚Üí    message: hasInternalLinks
   659‚Üí      ? `${totalInternalLinks} interne link(s) gevonden in de inhoud.`
   660‚Üí      : linkSuggestionAvailable
   661‚Üí        ? 'Geen interne links gevonden. Overweeg links toe te voegen naar bestaande pagina\'s.'
   662‚Üí        : 'Geen bestaande pagina\'s beschikbaar om naar te linken.',
   663‚Üí    details: { totalInternalLinks, suggestions },
   664‚Üí  };
   665‚Üí}

   671‚Üí/**
   672‚Üí * Result of running all quality gates on a piece of generated content.
   673‚Üí */
   674‚Üíexport interface QualityGateRunResult {
   675‚Üí  /** Individual results for each gate */
   676‚Üí  results: QualityGateResult[];
   677‚Üí  /** Average score across all gates (0-100) */
   678‚Üí  overallScore: number;
   679‚Üí  /** Whether the page passes ALL blocking gates */
   680‚Üí  approved: boolean;
   681‚Üí  /** List of rejection reasons (Dutch) for blocking gate failures */
   682‚Üí  rejectionReasons: string[];
   683‚Üí  /** List of warnings (Dutch) for non-blocking gate failures */
   684‚Üí  warnings: string[];
   685‚Üí}
   686‚Üí
   687‚Üí/**
   688‚Üí * Run all quality gates on generated content.
   689‚Üí *
   690‚Üí * The overall quality score is the average of all gate scores.
   691‚Üí * Pages that fail ANY blocking gate (duplicate, template completeness, brand check)
   692‚Üí * are rejected. Pages that fail non-blocking gates get warnings.
   693‚Üí *
   694‚Üí * @param projectId - The project context
   695‚Üí * @param templateId - The template being used
   696‚Üí * @param generatedContent - The generated content to check
   697‚Üí * @param rowData - The data row used to generate this content
   698‚Üí * @param config - Optional quality gate configuration overrides
   699‚Üí * @returns Comprehensive quality gate results
   700‚Üí */
   701‚Üíexport async function runQualityGates(
   702‚Üí  projectId: string,
   703‚Üí  templateId: string,
   704‚Üí  generatedContent: string,
   705‚Üí  rowData: Record<string, unknown>,
   706‚Üí  config?: QualityGatesConfig
   707‚Üí): Promise<QualityGateRunResult> {
   708‚Üí  const qualityConfig: QualityGatesConfig = config ?? {
   709‚Üí    minWordCount: 300,
   710‚Üí    minUniqueDataPoints: 3,
   711‚Üí    maxSimilarityThreshold: 0.8,
   712‚Üí    checkCannibalisation: true,
   713‚Üí    checkBrandCompliance: true,
   714‚Üí    checkClaims: true,
   715‚Üí    requireInternalLinks: true,
   716‚Üí  };
   717‚Üí
   718‚Üí  // Get the template to determine key variables
   719‚Üí  const template = await db.programmaticTemplate.findUnique({
   720‚Üí    where: { id: templateId },
   721‚Üí  });
   722‚Üí
   723‚Üí  const variables: Array<{ name: string; required: boolean }> = template?.variables
   724‚Üí    ? JSON.parse(template.variables)
   725‚Üí    : [];
   726‚Üí
   727‚Üí  const keyVariables = variables.filter((v) => v.required).map((v) => v.name);
   728‚Üí
   729‚Üí  // Determine the target keyword from the rendered content or row data
   730‚Üí  const targetKeyword = extractTargetKeyword(rowData, template?.targetKeyword);
   731‚Üí
   732‚Üí  // Run all quality gates
   733‚Üí  const results: QualityGateResult[] = [];
   734‚Üí
   735‚Üí  // Gate 1: Unique Data Requirement
   736‚Üí  results.push(await checkUniqueData(templateId, rowData, keyVariables));
   737‚Üí
   738‚Üí  // Gate 2: Minimum Value Threshold
   739‚Üí  results.push(checkMinValueThreshold(generatedContent, rowData, qualityConfig));
   740‚Üí
   741‚Üí  // Gate 3: Duplicate Check (BLOCKING)
   742‚Üí  results.push(await checkDuplicate(projectId, generatedContent, qualityConfig));
   743‚Üí
   744‚Üí  // Gate 4: Cannibalisation Check
   745‚Üí  if (qualityConfig.checkCannibalisation !== false) {
   746‚Üí    results.push(await checkCannibalisation(projectId, targetKeyword));
   747‚Üí  }
   748‚Üí
   749‚Üí  // Gate 5: Template Completeness (BLOCKING)
   750‚Üí  results.push(checkTemplateCompleteness(generatedContent, rowData));
   751‚Üí
   752‚Üí  // Gate 6: Brand Check (BLOCKING)
   753‚Üí  if (qualityConfig.checkBrandCompliance !== false) {
   754‚Üí    results.push(await checkBrandCompliance(projectId, generatedContent));
   755‚Üí  }
   756‚Üí
   757‚Üí  // Gate 7: Claim Check
   758‚Üí  if (qualityConfig.checkClaims !== false) {
   759‚Üí    results.push(checkClaims(generatedContent));
   760‚Üí  }
   761‚Üí
   762‚Üí  // Gate 8: Internal Link Check
   763‚Üí  if (qualityConfig.requireInternalLinks !== false) {
   764‚Üí    results.push(await checkInternalLinks(projectId, generatedContent));
   765‚Üí  }
   766‚Üí
   767‚Üí  // Calculate overall score
   768‚Üí  const overallScore =
   769‚Üí    results.length > 0
   770‚Üí      ? Math.round(results.reduce((sum, r) => sum + r.score, 0) / results.length)
   771‚Üí      : 0;
   772‚Üí
   773‚Üí  // Determine blocking gate failures
   774‚Üí  const blockingGateIds = new Set([
   775‚Üí    QUALITY_GATES.DUPLICATE.id,
   776‚Üí    QUALITY_GATES.TEMPLATE_COMPLETENESS.id,
   777‚Üí    QUALITY_GATES.BRAND_CHECK.id,
   778‚Üí  ]);
   779‚Üí
   780‚Üí  const rejectionReasons: string[] = [];
   781‚Üí  const warnings: string[] = [];
   782‚Üí
   783‚Üí  for (const result of results) {
   784‚Üí    if (!result.passed) {
   785‚Üí      // Check if this is a blocking gate by matching the gate name
   786‚Üí      const gateEntry = Object.values(QUALITY_GATES).find((g) => g.name === result.gateName);
   787‚Üí      if (gateEntry && gateEntry.blocking) {
   788‚Üí        rejectionReasons.push(result.message);
   789‚Üí      } else {
   790‚Üí        warnings.push(result.message);
   791‚Üí      }
   792‚Üí    }
   793‚Üí  }
   794‚Üí
   795‚Üí  const approved = rejectionReasons.length === 0;
   796‚Üí
   797‚Üí  return {
   798‚Üí    results,
   799‚Üí    overallScore,
   800‚Üí    approved,
   801‚Üí    rejectionReasons,
   802‚Üí    warnings,
   803‚Üí  };
   804‚Üí}

   806‚Üí/**
   807‚Üí * Extract the target keyword from row data by rendering the keyword pattern.
   808‚Üí */
   809‚Üífunction extractTargetKeyword(
   810‚Üí  rowData: Record<string, unknown>,
   811‚Üí  keywordPattern?: string | null
   812‚Üí): string {
   813‚Üí  if (!keywordPattern) return '';
   814‚Üí
   815‚Üí  let keyword = keywordPattern;
   816‚Üí  for (const [key, value] of Object.entries(rowData)) {
   817‚Üí    const placeholder = new RegExp(`\\{\\{${key}\\}\\}`, 'g');
   818‚Üí    keyword = keyword.replace(placeholder, String(value));
   819‚Üí  }
   820‚Üí  return keyword;
   821‚Üí}
